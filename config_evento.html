async function salvarEvento() { const btn = document.querySelector('.btn-save'); // ou o ID do seu botão btn.disabled = true; btn.innerText = "SALVANDO..."; const id = document.getElementById('input-id').value || Date.now().toString(); const nome = document.getElementById('input-nome').value.trim().replace(/\s+/g, '_'); const tempo = document.getElementById('input-tempo').value; // Pegue as URLs que já foram enviadas para o R2 (Música e Moldura) const musicaUrl = document.getElementById('url-musica').value; const molduraUrl = document.getElementById('url-moldura').value; const dados = { id: id, nome: nome, tempo: tempo, musicaUrl: musicaUrl, molduraUrl: molduraUrl, data: new Date().toLocaleDateString('pt-BR') }; try { const res = await fetch(`${WORKER_URL}/salvar-evento`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(dados) }); if (res.ok) { alert("Evento Salvo com Sucesso!"); window.location.href = 'index.html'; // Volta para a central } else { alert("Erro ao salvar no banco de dados."); } } catch (e) { alert("Erro de conexão com o Worker."); } finally { btn.disabled = false; btn.innerText = "SALVAR EVENTO"; } }
